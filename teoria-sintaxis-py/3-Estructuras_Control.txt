#Bucle/loops
#FOR
frutas = ["manzana", "banana", "naranja"]

for fruta in frutas: #"Repite hasta terminar la secuencia/cadena"
    print(fruta) 


for i in range(10):
    print(i)

#WHILE
contador = 0
while contador < 5:

    print(contador)
    contador += 1

BREAK: Corta el bucle y sigue con el codigo fuera de el
while True:

    print(contador)
    contador += 1


    if contador == 5:
        break

CONTINUE: continua con la siguiente iteracion, permite que no revise el resto del codigo si no cumple
for i in range(10):

PASS: Permite pasar toda la iteracion ssin hacer nada, sirve para una posterior implementacion del bucle
for i in range(5):
    pass

    if i % 2 == 0:
        continue
    print(i)
